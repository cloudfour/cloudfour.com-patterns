import { Meta } from '@storybook/addon-docs/blocks';
import tokens from '../compiled/tokens/js/tokens';
import _ from 'lodash';
const opacities = tokens.number.opacity;
const opacityRows = Object.keys(opacities).map((key) => {
  const { name, attributes, value } = opacities[key];
  return (
    <tr key={name}>
      <td>opacity.${_.kebabCase(attributes.item)}</td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  );
});
