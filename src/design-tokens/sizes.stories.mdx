import { Meta } from '@storybook/addon-docs/blocks';
import * as msTokens from './modular-scale.yml';
import * as sizeTokens from './sizes.yml';
const msStepRows = [];
for (let i = -4; i <= 10; i++) {
  const msStep = msTokens.ratio ** i;
  const msStepRow = (
    <tr>
      <th scope="row">{i}</th>
      <td>{msStep.toFixed(2)}</td>
    </tr>
  );
  msStepRows.push(msStepRow);
}
const sizeTokenRows = Object.keys(sizeTokens).map(key => {
  return (
    <tr>
      <th scope="row">{key}</th>
      <td>{sizeTokens[key]}</td>
    </tr>
  );
});
