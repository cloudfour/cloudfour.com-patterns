import { ColorItem, ColorPalette, Meta } from '@storybook/addon-docs';
import tokens from '../compiled/tokens/js/tokens.js';
const generateColorPalette = (props) => {
  const propsArray = Object.values(props);
  return (
    <ColorPalette>
      {propsArray.map((prop) => (
        <ColorItem
          key={prop.name}
          title={`color.$${prop.name.slice(6)}`}
          subtitle={prop.comment}
          colors={[prop.value]}
        />
      ))}
    </ColorPalette>
  );
};
const brandColorPalette = generateColorPalette(tokens.color.brand);
const baseColorPalette = generateColorPalette(tokens.color.base);
const textColorPalette = generateColorPalette(tokens.color.text);
