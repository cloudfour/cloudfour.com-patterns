import * as msTokens from './modular-scale.yml';
import * as sizeTokens from './sizes.yml';
let msStepRows = [];
for (let i = -4; i <= 10; i++) {
  let msStep = Math.pow(msTokens.ratio, i);
  let msStepRow = <tr><th scope="row">{i}</th><td>{msStep.toFixed(2)}</td></tr>;
  msStepRows.push(msStepRow);
}
const sizeTokenRows = Object.keys(sizeTokens).map(key => {
  return <tr><th scope="row">{key}</th><td>{sizeTokens[key]}</td></tr>;
});
