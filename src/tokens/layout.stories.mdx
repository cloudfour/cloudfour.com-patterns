import { Meta } from '@storybook/addon-docs/blocks';
import tokens from '../compiled/tokens/js/tokens';
const maxWidths = tokens.size.max_width;
const maxWidthRows = Object.keys(maxWidths).map((key) => {
  const { name, attributes, value } = maxWidths[key];
  return (
    <tr key={name}>
      <td>
        <code>size.${name.slice(attributes.category.length + 1)}</code>
      </td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  );
});
