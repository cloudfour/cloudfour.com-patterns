import { Meta } from '@storybook/addon-docs';
import tokens from '../compiled/tokens/js/tokens.js';
const _ = require('lodash');
const breakpoints = tokens.size.breakpoint;
const breakpointRows = Object.keys(breakpoints).map((key) => {
  const { name, attributes, value } = breakpoints[key];
  return (
    <tr key={name}>
      <td>
        <code>breakpoint.${_.kebabCase(attributes.item)}</code>
      </td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  );
});
