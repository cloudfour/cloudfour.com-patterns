import { Meta } from '@storybook/addon-docs';
import tokens from '../compiled/tokens/js/tokens';
const _ = require('lodash');
const modularScale = tokens.number.scale.modular;
const modularScaleRows = Object.keys(modularScale).map((key) => {
  const { name, attributes, value } = modularScale[key];
  return (
    <tr key={name}>
      <td>
        <code>
          scale.${_.kebabCase(`${attributes.item}-${attributes.subitem}`)}
        </code>
      </td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  );
});
