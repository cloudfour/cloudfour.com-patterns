import { Meta } from '@storybook/addon-docs/blocks';
import tokens from '../compiled/js/tokens';
const maxWidths = tokens.size.max_width;
const maxWidthRows = Object.keys(maxWidths).map((key) => {
  const { name, value } = maxWidths[key];
  return (
    <tr key={name}>
      <td>
        <code>${name}</code>
      </td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  );
});
